import os

def scan_ports(host):
    for port in range(1,1024):
        result = subprocess.run(['nc','-zv',host,str(port)], capture_output=True)
        if result.returncode == 0:
            print(f"Port {port} is open")

def list_processes():
    for pid in os.listdir('/proc'):
        if pid.isdigit():
            with open(f'/proc/{pid}/cmdline','r') as f:
                cmd = f.read()
                if cmd:
                    print(f"PID {pid}: {cmd}")
