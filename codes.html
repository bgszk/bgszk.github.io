C++

#include <iostream>
#include <vector>

class Enemy {
    int health;
public:
    Enemy(int h) : health(h) {}
    void takeDamage(int d) { health -= d; }
    int getHealth() { return health; }
};

int main() {
    std::vector<Enemy> enemies = {Enemy(100), Enemy(150)};
    enemies[0].takeDamage(40);
    std::cout << "Inimigo 0 vida: " << enemies[0].getHealth() << std::endl;
}
