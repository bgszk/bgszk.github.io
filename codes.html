// Manipulação de memória e ponteiros
#include <stdio.h>
#include <stdlib.h>

struct KernelHook {
    void* target;
    void (*callback)();
};

void override(void* addr) {
    struct KernelHook hook;
    hook.target = addr;
    hook.callback = [](){ printf("Kernel function overridden\n"); };
    printf("Memory patched at %p\n", hook.target);
}

int main() {
    void* fakeAddr = (void*)0xDEADBEEF;
    override(fakeAddr);
    return 0;
}

// Low-level safe systems
fn main() {
    let mut buffer: [u8; 1024] = [0; 1024];
    unsafe {
        let ptr = buffer.as_mut_ptr();
        for i in 0..buffer.len() {
            *ptr.add(i) = (i % 256) as u8;
        }
    }
    println!("Buffer initialized and manipulated");
}
