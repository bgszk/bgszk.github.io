#include <stdio.h>
#include <stdint.h>
#include <string.h>

typedef struct {
    uint32_t eax;
    uint32_t ebx;
    uint32_t ecx;
    uint32_t edx;
} cpu_state_t;

static void dump_state(const cpu_state_t *state) {
    printf("EAX=%08X\n", state->eax);
    printf("EBX=%08X\n", state->ebx);
    printf("ECX=%08X\n", state->ecx);
    printf("EDX=%08X\n", state->edx);
}

static void xor_block(uint8_t *buf, size_t len, uint8_t key) {
    for (size_t i = 0; i < len; ++i) {
        buf[i] ^= key;
    }
}

int main(void) {
    cpu_state_t state = {
        .eax = 0xDEADBEEF,
        .ebx = 0xCAFEBABE,
        .ecx = 0x41414141,
        .edx = 0x13371337
    };

    uint8_t payload[] = "raw-buffer-control";
    xor_block(payload, strlen((char*)payload), 0x5A);

    dump_state(&state);
    printf("patched: ");
    for (size_t i = 0; i < sizeof(payload) - 1; ++i) {
        printf("%02X ", payload[i]);
    }
    printf("\n");

    return 0;
}

#include <iostream>
#include <vector>
#include <memory>
#include <algorithm>
#include <cstdint>

class MemoryRegion {
public:
    MemoryRegion(std::uintptr_t base, std::size_t size)
        : base_(base), size_(size) {}

    bool contains(std::uintptr_t addr) const {
        return addr >= base_ && addr < (base_ + size_);
    }

    void dump() const {
        std::cout << "region: 0x" << std::hex << base_
                  << " size: 0x" << size_ << std::dec << '\n';
    }

private:
    std::uintptr_t base_;
    std::size_t size_;
};

int main() {
    std::vector<std::unique_ptr<MemoryRegion>> regions;
    regions.emplace_back(std::make_unique<MemoryRegion>(0x1000, 0x200));
    regions.emplace_back(std::make_unique<MemoryRegion>(0x2000, 0x400));
    regions.emplace_back(std::make_unique<MemoryRegion>(0x3000, 0x800));

    for (const auto& region : regions) {
        region->dump();
    }

    auto target = std::find_if(
        regions.begin(),
        regions.end(),
        [](const auto& region) {
            return region->contains(0x22AF);
        }
    );

    if (target != regions.end()) {
        std::cout << "address mapped inside region\n";
    } else {
        std::cout << "address not mapped\n";
    }

    return 0;
}

using System;
using System.Linq;
using System.Reflection;
using System.Collections.Generic;

public class HookRegistry
{
    private readonly Dictionary<string, MethodInfo> _methods = new();

    public void Scan(Type t)
    {
        foreach (var method in t.GetMethods(
            BindingFlags.Public |
            BindingFlags.NonPublic |
            BindingFlags.Static |
            BindingFlags.Instance))
        {
            _methods[method.Name] = method;
        }
    }

    public void Dump()
    {
        foreach (var pair in _methods.OrderBy(p => p.Key))
        {
            Console.WriteLine($"{pair.Key} -> {pair.Value.ReturnType.Name}");
        }
    }
}

public class TargetRuntime
{
    private static string ResolveToken(int token) => $"token:{token:X}";
    public int Patch(int address) => address ^ 0x45AF;
}

public static class Program
{
    public static void Main()
    {
        var registry = new HookRegistry();
        registry.Scan(typeof(TargetRuntime));
        registry.Dump();

        var runtime = new TargetRuntime();
        Console.WriteLine(runtime.Patch(0x401000));
    }
}

use std::collections::HashMap;

#[derive(Debug)]
struct Section {
    name: String,
    base: usize,
    size: usize,
}

impl Section {
    fn contains(&self, addr: usize) -> bool {
        addr >= self.base && addr < self.base + self.size
    }
}

fn main() {
    let mut table: HashMap<&str, Section> = HashMap::new();

    table.insert("text", Section {
        name: ".text".into(),
        base: 0x401000,
        size: 0x1800,
    });

    table.insert("rdata", Section {
        name: ".rdata".into(),
        base: 0x403000,
        size: 0x0C00,
    });

    table.insert("data", Section {
        name: ".data".into(),
        base: 0x404000,
        size: 0x0800,
    });

    let probe = 0x401ABC;

    for (_, section) in &table {
        if section.contains(probe) {
            println!("address {:#X} inside {}", probe, section.name);
        }
    }
}

class BinaryInspector
  attr_reader :path, :strings

  def initialize(path)
    @path = path
    @strings = []
  end

  def extract_ascii(min = 4)
    data = File.binread(path)
    current = +""

    data.each_byte do |byte|
      if byte.between?(32, 126)
        current << byte.chr
      else
        if current.length >= min
          @strings << current.dup
        end
        current.clear
      end
    end

    @strings
  end

  def summary
    puts "file: #{path}"
    puts "strings found: #{@strings.length}"
    @strings.first(10).each_with_index do |str, i|
      puts "[#{i}] #{str}"
    end
  end
end

inspector = BinaryInspector.new("sample.bin")
inspector.extract_ascii
inspector.summary

#!/usr/bin/env bash

set -euo pipefail

TARGET_DIR="${1:-/usr/bin}"
TMP_FILE="/tmp/scan_report.txt"

echo "[*] scanning: ${TARGET_DIR}" > "${TMP_FILE}"

find "${TARGET_DIR}" -maxdepth 1 -type f 2>/dev/null | while read -r file; do
  perms=$(stat -c "%A" "$file" 2>/dev/null || true)
  size=$(stat -c "%s" "$file" 2>/dev/null || true)
  mime=$(file -b "$file" 2>/dev/null || true)

  printf "%s | %s bytes | %s | %s\n" \
    "$file" \
    "$size" \
    "$perms" \
    "$mime" \
    >> "${TMP_FILE}"
done

sort "${TMP_FILE}" | tail -n 50

echo "[+] report saved in ${TMP_FILE}"
